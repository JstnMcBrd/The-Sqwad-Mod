M0N_conquest = {
	slot = 1 #?
	generic = no
	ai = yes
	has_country_shield = yes
	
	potential = {
		tag = M0N
	}
	
	M0N_mighty_army = {
		icon = mission_assemble_an_army
		required_missions = { }
		position = 1
		ai_weight = { factor = 5 }
		ai_priority = { factor = 5 }
		
		trigger = {
			army_size_percentage = 0.8
			#manpower_percentage = 0.5
			num_of_generals = 1
		}
		
		effect = {
			add_prestige = 10		
			#Territorial Claims
			if = {
				limit = { NOT = { map_setup = map_setup_random } }
				carinthia_area = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
				transdanubia_area = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
				moravia_area = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
				lorraine_area = {
					limit = {
							NOT = { is_core = ROOT }
							NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
			if = { #if map is set on random, add claims on bordering areas
				limit = { map_setup = map_setup_random }
				
				every_owned_province = {
					every_neighbor_province = {
						area = {
							limit = {
								NOT = { is_core = ROOT }
								NOT = { is_permanent_claim = ROOT }
							}
							add_permanent_claim = ROOT
						}
					}
				}
			}		
		}
	}
}

		required_missions = { M0N_mighty_army }
		position = 6
		ai_weight = { factor = 5 }
		ai_priority = { factor = 5 }
		
		provinces_to_highlight = {
			area = carinthia_area
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		
		trigger = {
			carinthia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		
		effect = {
			add_prestige = 10
			add_country_modifier = {
				name = "sqwad_territorial_gains_modifier"

				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			#Claims on Venetia to start Italian conquest tree
			venetia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
