				limit = {
					NOT = { is_core = ROOT }

				}
				add_permanent_claim = ROOT
			}
			#Southeast France Coast claims
			savoy_dauphine_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}	
			languedoc_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			if = {
				limit = {
					exists = FRA
					FRA = {
						OR = {
							any_owned_province = {
								OR = { 
									area = savoy_dauphine_area
									area = languedoc_area
								}
							}
							any_subject_country = {
								any_owned_province = {
									OR = { 
										area = savoy_dauphine_area
										area = languedoc_area
									}
								}
							}
						}
					}
				}
				add_casus_belli = {
					target = FRA
					type = cb_vassalize_mission
				}
			}			
		}
	}
}
